Page Numbering Controls

Usage Example

// Create a Previewer with page numbering configuration
const previewer = new PaperView.Previewer({ viewMode: "single" });

previewer.preview(sourceContent, renderToElement, ['assets/styles/paper.css']).then((flow) => {
    // Enable page numbering after rendering
    flow.setPageNumberingEnabled(true);

    // Update configuration dynamically
    flow.updatePageNumbering({
        position: 'top-right',
        style: 'upper-roman'
    });
});
